#include <stdio.h>
#include <ctype.h>  // for isspace()

int countWords(const char *str) {
    int count = 0;
    int inWord = 0;

    while (*str) {
        if (!isspace(*str)) {
            if (!inWord) {
                count++;
                inWord = 1;
            }
        } else {
            inWord = 0;
        }
        str++;
    }

    return count;
}

int main() {
    char str[] = "Count how many words are in this sentence.";
    int wordCount = countWords(str);

    printf("Original string: \"%s\"\n", str);
    printf("Word count: %d\n", wordCount);

    return 0;
}
