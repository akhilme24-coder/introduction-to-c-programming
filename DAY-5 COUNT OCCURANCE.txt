#include <stdio.h>
int main()
{
int arr[100],n,i,x,count=0;
printf("enter number of elements : ");
scanf("%d",&n);
printf("enter %d element :\n",n);
for(i=0;i<n;i++)
{
scanf("%d",&arr[i]);
}
printf("enter elements to count : \n");
scanf("%d",&x);
for(i=0;i<n;i++){
if(arr[i]==x)
count++;
}
printf("%d occurs %d time \n",x,count);
return 0;
}
