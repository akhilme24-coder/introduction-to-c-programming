#include <stdio.h>
#include <string.h>

int main() {
    char numStr[100], revStr[100];
    int length, i;

    printf("Enter a number: ");
    scanf("%s", numStr);

    length = strlen(numStr);

    // Reverse the string
    for(i = 0; i < length; i++) {
        revStr[i] = numStr[length - 1 - i];
    }
    revStr[length] = '\0';  // Null-terminate the reversed string

    printf("Reversed number: %s\n", revStr);

    return 0;
}
